import tkinter as tk
from tkinter import messagebox

def updateJiraClick():
    # Create the root window but don't show it
    root = tk.Tk()
    root.withdraw()

    # Create and display a messagebox asking for confirmation
    response = messagebox.askyesno("Confirmation", "Do you really want to update?")

    # Check user's response
    if response:
        # Code to update JIRA goes here
        print("Updating JIRA...")
    else:
        print("Update cancelled.")

    # Destroy the root window
    root.destroy()
